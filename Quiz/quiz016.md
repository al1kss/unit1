```.py
def get_l3tt3r(msg):
    result = ""
    for letter in msg:
        if letter == "o":
            result += "0"
        elif letter == "e":
            result += "3"
        elif letter == "a":
            result += "4"
        elif letter == "i":
            result += "1"
        elif letter == " ":
            result += "_"
        else:
            result += letter
    return result
print(get_l3tt3r(msg = "Hello World"))
print(get_l3tt3r(msg = "Why did I choose CS?"))
print(get_l3tt3r(msg = "Remember the Figure Caption"))

```
